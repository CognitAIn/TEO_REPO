PS C:\Users\user\Desktop\Trinity_STEM> function Show-Progress {
    param([string]$phase, [int]$total, [int]$interval)
    $elapsed = 0
    while ($elapsed -lt $total) {
        $cpu = [math]::Round((Get-Counter '\Processor(_Total)\% Processor Time').CounterSamples.CookedValue, 1)
        $mem = [math]::Round((Get-Counter '\Memory\Available MBytes').CounterSamples.CookedValue, 1)
        Write-Host ("[{0}] ⏱ {1,4}s | CPU {2,5:N1}% | MEM Avail {3,6:N1} MB" -f $phase, $elapsed, $cpu, $mem)
        Start-Sleep -Seconds $interval
        $elapsed += $interval
    }
}
